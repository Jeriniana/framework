package framework ;

import java.lang.Class ;
import java.util.Set ;
import java.util.HashSet ;
import java.net.URL ;
import java.util.Enumeration ;
import java.io.File ;

public class PackageScanner {
    public Set<String> findAnnotedClasses (String annotation, String packageName) {
        Set<String> result = new HashSet<String>();

        try {
            ClassLoader classloader = Thread.currentThread().getContextClassLoader();
            String packagePath = packageName.replace('.', '/');
            Enumeration<URL> resources = classloader.getResources(packagePath);
            while ( resources.hasMoreElements()) {
                URL resource = resources.nextElement();
                File dir = new File (resource.getFile());

                Set<String> annotedClasses = findAnnotedClassesFromDir (annotation, packageName, dir);
                result.addAll(annotedClasses);

            }
        } catch ( Exception e ) {
            System.out.println (e.getMessage());
        }

        return result ;

    }

    public Set<String> findAnnotedClassesFromDir (String annotation, String packageName, File dir) {
        Set<String> annotedClasses = new HashSet<String>();

        File [] files = dir.listFiles();
        if (files == null) return annotedClasses ;

            if (f.isDirectory()) {
                
                String nextPackageName = packageName.isEmpty() ? f.getName() : packageName + "." + f.getName();
